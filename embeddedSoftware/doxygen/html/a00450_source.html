82 void MDB_GPIO_Init(void) {
83  GPIO_InitTypeDef GPIO_InitStructure;
84  MD_GPIO_NAME_t name;
85 
86  for(name=(MD_GPIO_NAME_t)0; name < (MD_GPIO_NAME_t)GPIO_COUNT; name++) {
88  RCC_AHB1PeriphClockCmd(MDB_GPIO[name]._GPIO_AHB_CLK, ENABLE);
89 
90  GPIO_InitStructure.GPIO_Pin = MDB_GPIO[name]._GPIO_PIN;
91  GPIO_InitStructure.GPIO_Mode = MDB_GPIO[name]._GPIO_Mode;
92  GPIO_InitStructure.GPIO_OType = MDB_GPIO[name]._GPIO_OType;
93  GPIO_InitStructure.GPIO_PuPd = MDB_GPIO[name]._GPIO_PuPd;
94  GPIO_InitStructure.GPIO_Speed = MDB_GPIO[name]._GPIO_Speed;
95  GPIO_Init(MDB_GPIO[name]._GPIO_PORT, &GPIO_InitStructure);
96 
97  // Default Wert einstellen
98  if(MDB_GPIO[name]._GPIO_INIT==GPIO_OFF) {
99  MDB_GPIO_Off(name);
100  }
101  else {
102  MDB_GPIO_On(name);
103  }
104  }
