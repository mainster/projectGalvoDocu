220 void QSPI_Init(QSPI_InitTypeDef* QSPI_InitStruct)
221 {
222  uint32_t tmpreg = 0;
223 
224  /* Check the QSPI parameters */
225  assert_param(IS_QSPI_SSHIFT(QSPI_InitStruct->QSPI_SShift));
226  assert_param(IS_QSPI_PRESCALER(QSPI_InitStruct->QSPI_Prescaler));
227  assert_param(IS_QSPI_CKMODE(QSPI_InitStruct->QSPI_CKMode));
228  assert_param(IS_QSPI_CSHTIME(QSPI_InitStruct->QSPI_CSHTime));
229  assert_param(IS_QSPI_FSIZE(QSPI_InitStruct->QSPI_FSize));
230  assert_param(IS_QSPI_FSEL(QSPI_InitStruct->QSPI_FSelect));
231  assert_param(IS_QSPI_DFM(QSPI_InitStruct->QSPI_DFlash));
232 
233  /*------------------------ QSPI CR Configuration ------------------------*/
234  /* Get the QUADSPI CR1 value */
235  tmpreg = QUADSPI->CR;
236  /* Clear PRESCALER and SSHIFT bits */
237  tmpreg &= QSPI_CR_CLEAR_MASK;
238  /* Configure QUADSPI: Prescaler and Sample Shift */
239  tmpreg |= (uint32_t)(((QSPI_InitStruct->QSPI_Prescaler)<<24)
240  |(QSPI_InitStruct->QSPI_SShift)
241  |(QSPI_InitStruct->QSPI_FSelect)
242  |(QSPI_InitStruct->QSPI_DFlash));
243  /* Write to QUADSPI CR */
244  QUADSPI->CR = tmpreg;
245 
246  /*------------------------ QUADSPI DCR Configuration ------------------------*/
247  /* Get the QUADSPI DCR value */
248  tmpreg = QUADSPI->DCR;
249  /* Clear FSIZE, CSHT and CKMODE bits */
250  tmpreg &= QSPI_DCR_CLEAR_MASK;
251  /* Configure QSPI: Flash Size, Chip Select High Time and Clock Mode */
252  tmpreg |= (uint32_t)(((QSPI_InitStruct->QSPI_FSize)<<16)
253  |(QSPI_InitStruct->QSPI_CSHTime)
254  |(QSPI_InitStruct->QSPI_CKMode));
255  /* Write to QSPI DCR */
256  QUADSPI->DCR = tmpreg;
257 }
