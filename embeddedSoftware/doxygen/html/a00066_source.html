109 uint8_t TM_USART_DMA_Send(USART_TypeDef* USARTx, uint8_t* DataArray, uint16_t count) {
110  /* Get USART settings */
111  TM_USART_DMA_INT_t* Settings = TM_USART_DMA_INT_GetSettings(USARTx);
112 
113  /* Check if DMA is working now */
114  if (Settings->DMA_Stream->NDTR) {
115  /* DMA works right now */
116  return 0;
117  }
118 
119  /* Set DMA options */
120  DMA_InitStruct.DMA_Channel = Settings->DMA_Channel;
121  DMA_InitStruct.DMA_BufferSize = count;
122  DMA_InitStruct.DMA_PeripheralBaseAddr = (uint32_t) &USARTx->DR;
123  DMA_InitStruct.DMA_Memory0BaseAddr = (uint32_t) &DataArray[0];
124 
125  /* Deinit first, clear all flags */
126  DMA_DeInit(Settings->DMA_Stream);
127 
128  /* Init DMA */
129  DMA_Init(Settings->DMA_Stream, &DMA_InitStruct);
130 
131  /* Enable USART TX DMA */
132  USARTx->CR3 |= USART_CR3_DMAT;
133 
134  /* Enable DMA Stream */
135  Settings->DMA_Stream->CR |= DMA_SxCR_EN;
136 
137  /* DMA has started */
138  return 1;
139 }
