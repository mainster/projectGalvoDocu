67 TM_TIMER_PROPERTIES_Result_t TM_TIMER_PROPERTIES_GenerateDataForWorkingFrequency(TM_TIMER_PROPERTIES_t* Timer_Data, double frequency) {
68  if (frequency > Timer_Data->TimerFrequency) {
69  /* Reset values */
70  Timer_Data->Prescaler = 0;
71  Timer_Data->Period = 0;
72  Timer_Data->Frequency = 0;
73 
74  /* Frequency too high */
75  return TM_TIMER_PROPERTIES_Result_FrequencyTooHigh;
76  } else if (frequency == 0) {
77  /* Reset values */
78  Timer_Data->Prescaler = 0;
79  Timer_Data->Period = 0;
80  Timer_Data->Frequency = 0;
81 
82  /* Not valid frequency */
83  return TM_TIMER_PROPERTIES_Result_FrequencyTooLow;
84  }
85 
86  /* Fix for 16/32bit timers */
87  if (Timer_Data->MaxPeriod <= 0xFFFF) {
88  Timer_Data->MaxPeriod++;
89  }
90 
91  /* Get minimum prescaler and maximum resolution for timer */
92  Timer_Data->Prescaler = 0;
93  do {
94  /* Get clock */
95  Timer_Data->Period = (Timer_Data->TimerFrequency / (Timer_Data->Prescaler + 1));
96  /* Get period */
97  Timer_Data->Period = (Timer_Data->Period / frequency);
98  /* Increase prescaler value */
99  Timer_Data->Prescaler++;
100  } while (Timer_Data->Period > (Timer_Data->MaxPeriod) && Timer_Data->Prescaler <= (Timer_Data->MaxPrescaler + 1));
101  /* Check for too low frequency */
102  if (Timer_Data->Prescaler > (Timer_Data->MaxPrescaler + 1)) {
103  /* Reset values */
104  Timer_Data->Prescaler = 0;
105  Timer_Data->Period = 0;
106  Timer_Data->Frequency = 0;
107 
108  /* Prescaler too high, frequency is too low for use */
109  return TM_TIMER_PROPERTIES_Result_FrequencyTooLow;
110  }
111 
112  /* Set frequency */
113  Timer_Data->Frequency = frequency;
114 
115  /* Return OK */
116  return TM_TIMER_PROPERTIES_Result_Ok;
117 }
