504 static __INLINE void TM_USART_Putc(USART_TypeDef* USARTx, volatile char c) {
505  /* Check USART if enabled */
506  if ((USARTx->CR1 & USART_CR1_UE)) {
507  /* Wait to be ready, buffer empty */
508  USART_WAIT(USARTx);
509  /* Send data */
510  USARTx->DR = (uint16_t)(c & 0x01FF);
511  /* Wait to be ready, buffer empty */
512  USART_WAIT(USARTx);
513  }
514 }
