205 int TM_Timer2_config(FunctionalState TimerRun, FunctionalState IntOn, uint16_t peri) {
206 TIM_TimeBaseInitTypeDef timer_conf;
207 
208 // Enable TIM clock using RCC_APBxPeriphClockCmd(RCC_APBxPeriph_TIMx, ENABLE)
209  RCC_APB1PeriphClockCmd(RCC_APB1Periph_TIM2, ENABLE);
210 
211 
212 // Fill the TIM_TimeBaseInitStruct with the desired parameters.
213  timer_conf.TIM_Prescaler = 1;
214  timer_conf.TIM_CounterMode = TIM_CounterMode_Up;
215  timer_conf.TIM_Period = peri;
216  timer_conf.TIM_ClockDivision = TIM_CKD_DIV1;
217  timer_conf.TIM_RepetitionCounter = 0x00;
218 
219  /* Set Timer 2 Update trigger source enabled
220  * ADC start regular group scan is triggerd from that event */
221  TIM_SelectOutputTrigger(TIM2,TIM_TRGOSource_Update);
222 
223  // Call the TIM_Cmd(ENABLE) function to enable the TIM counter.
224  TIM_Cmd(TIM2, TimerRun);
225 
226 // Call TIM_TimeBaseInit(TIMx, &TIM_TimeBaseInitStruct) to configure the Time Base unit
227 // with the corresponding configuration
228  TIM_TimeBaseInit(TIM2, &timer_conf);
229 
230 // Enable the corresponding interrupt using the function TIM_ITConfig(TIMx, TIM_IT_Update)
231  TIM_ITConfig(TIM2, TIM_IT_Update, IntOn);
232  /* Try this: Enable TIM_IT_Trigger source for IRQn request. Toggle debug
233  * timing pin and toggle again when dma transmission complete IRQ occures */
234  TIM_ITConfig(TIM2, TIM_IT_Trigger, IntOn);
235 
236  /* Enable nvic channel */
237  timer2_interrupt_cmd(IntOn);
238 
239  return 0;
240 }
