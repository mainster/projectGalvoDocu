197 uint8_t TM_USART_Getc(USART_TypeDef* USARTx) {
198  int8_t c = 0;
199  TM_USART_t* u = TM_USART_INT_GetUsart(USARTx);
200 
201  /* Check if we have any data in buffer */
202  if (u->Num > 0 || u->In != u->Out) {
203  /* Check overflow */
204  if (u->Out == u->Size) {
205  u->Out = 0;
206  }
207 
208  /* Read character */
209  c = u->Buffer[u->Out];
210 
211  /* Increase output pointer */
212  u->Out++;
213 
214  /* Decrease number of elements */
215  if (u->Num) {
216  u->Num--;
217  }
218  }
219 
220  /* Return character */
221  return c;
222 }

224 uint16_t TM_USART_Gets(USART_TypeDef* USARTx, char* buffer, uint16_t bufsize) {
225  uint16_t i = 0;
226 
227  /* Get USART structure */
228  TM_USART_t* u = TM_USART_INT_GetUsart(USARTx);
229 
230  /* Check for any data on USART */
231  if (
232  u->Num == 0 ||
233  (
234  !TM_USART_FindCharacter(USARTx, u->StringDelimiter) &&
235  u->Num != u->Size
236  )
237  ) {
238  /* Return 0 */
239  return 0;
240  }
241 
242  /* If available buffer size is more than 0 characters */
243  while (i < (bufsize - 1)) {
244  /* We have available data */
245  buffer[i] = (char) TM_USART_Getc(USARTx);
246 
247  /* Check for end of string */
248  if ((uint8_t) buffer[i] == (uint8_t) u->StringDelimiter) {
249  /* Done */
250  break;
251  }
252 
253  /* Increase */
254  i++;
255  }
256 
257  /* Add zero to the end of string */
258  buffer[++i] = 0;
259 
260  /* Return number of characters in buffer */
261  return i;
262 }

295 uint8_t TM_USART_FindCharacter(USART_TypeDef* USARTx, uint8_t c) {
296  uint16_t num, out;
297  TM_USART_t* u = TM_USART_INT_GetUsart(USARTx);
298 
299  /* Temp variables */
300  num = u->Num;
301  out = u->Out;
302 
303  while (num > 0) {
304  /* Check overflow */
305  if (out == u->Size) {
306  out = 0;
307  }
308 
309  /* Check if characters matches */
310  if ((uint8_t) u->Buffer[out] == (uint8_t) c) {
311  /* Character found */
312  return 1;
313  }
314 
315  /* Set new variables */
316  out++;
317  num--;
318  }
319 
320  /* Character is not in buffer */
321  return 0;
322 }

324 void TM_USART_Puts(USART_TypeDef* USARTx, char* str) {
325  TM_USART_t* u = TM_USART_INT_GetUsart(USARTx);
326  /* If we are not initialized */
327  if (u->Initialized == 0) {
328  return;
329  }
330 
331  /* Go through entire string */
332  while (*str) {
333  /* Wait to be ready, buffer empty */
334  USART_WAIT(USARTx);
335  /* Send data */
336  USARTx->DR = (uint16_t)(*str++ & 0x01FF);
337  /* Wait to be ready, buffer empty */
338  USART_WAIT(USARTx);
339  }
340 }

363  /* Still available space in buffer */

365  /* Check overflow */

367  u->In = 0;
368  }
369 
370  /* Add to buffer */
371  u->Buffer[u->In] = c;
372  u->In++;
373  u->Num++;
374  }
